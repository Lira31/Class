Задание 4
num1 = int(input("Введите первое число"))
num2 = int(input("Введите второе число"))
a = input("Что сделать с этими числами?: умножить, разделить, сложить, вычесть")
if a == "умножить":
    print(num1*num2)
elif a == "разделить":
    print(num1//num2)
elif a == "сложить":
    print(num1+num2)
elif a == "вычесть":
    print(num1-num2)
else:
    print("Нет ответа")
